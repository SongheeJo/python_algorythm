```python
# [문제 출처] https://www.acmicpc.net/problem/11022

T = input()
T = int(T)
for i in range(T+1): # range는 연속되는 숫자를 list로 생성, [0, 1, 2, 3, .. T-1]
    if i == 0:
        pass
    else:
        A, B = map(int, input().split())
        print('Case #'+str(i)+': '+str(A)+' + '+str(B)+' = '+str(A+B))
        i += 1
```
